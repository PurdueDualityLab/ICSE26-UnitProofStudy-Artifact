49 inline int
50 snmp_ber_encode_unsigned_integer(snmp_packet_t *snmp_packet, uint8_t type, uint32_t number)
51 {
52 uint16_t original_out_len;
53
54 original_out_len = snmp_packet->used;
55 do {
56 if(snmp_packet->used == snmp_packet->max) {
57 return 0;
58 }
59
60 *snmp_packet->out-- = (uint8_t)number & 0xFF;
61 snmp_packet->used++;
62 /* I'm not sure why but on MSPGCC the >> 8 operation goes haywire here */
63 #ifdef __MSPGCC__
64 number >>= 4;
65 number >>= 4;
66 #else /* __MSPGCC__ */
67 number >>= 8;
68 #endif /* __MSPGCC__ */
69 } while(number);
70
71 if(!snmp_ber_encode_length(snmp_packet, snmp_packet->used - original_out_len)) {
72 return 0;
73 }
74
75 if(!snmp_ber_encode_type(snmp_packet, type)) {
76 return 0;
77 }
78
79 return 1;
80 }
81 /*---------------------------------------------------------------------------*/
82 int
83 snmp_ber_encode_type(snmp_packet_t *snmp_packet, uint8_t type)
84 {
85 if(snmp_packet->used == snmp_packet->max) {
86 return 0;
87 }
88
89 *snmp_packet->out-- = type;
90 snmp_packet->used++;
91
92 return 1;
93 }
94 /*---------------------------------------------------------------------------*/
95 int
96 snmp_ber_encode_length(snmp_packet_t *snmp_packet, uint16_t length)
97 {
98 if(length > 0xFF) {
99 if(snmp_packet->used == snmp_packet->max) {
100 return 0;
101 }
102
103 *snmp_packet->out-- = (uint8_t)length & 0xFF;
104 snmp_packet->used++;
105
106 if(snmp_packet->used == snmp_packet->max) {
107 return 0;
108 }
109
110 *snmp_packet->out-- = (uint8_t)(length >> 8) & 0xFF;
111 snmp_packet->used++;
112
113 if(snmp_packet->used == snmp_packet->max) {
114 return 0;
115 }
116
117 *snmp_packet->out-- = 0x82;
118 snmp_packet->used++;
119 } else if(length > 0x7F) {
120 if(snmp_packet->used == snmp_packet->max) {
121 return 0;
122 }
123
124 *snmp_packet->out-- = (uint8_t)length & 0xFF;
125 snmp_packet->used++;
126
127 if(snmp_packet->used == snmp_packet->max) {
128 return 0;
129 }
130
131 *snmp_packet->out-- = 0x81;
132 snmp_packet->used++;
133 } else {
134 if(snmp_packet->used == snmp_packet->max) {
135 return 0;
136 }
137
138 *snmp_packet->out-- = (uint8_t)length & 0x7F;
139 snmp_packet->used++;
140 }
141
142 return 1;
143 }

156 /*---------------------------------------------------------------------------*/
157 int
158 snmp_ber_encode_string_len(snmp_packet_t *snmp_packet, const char *str, uint32_t length)
159 {
160 uint32_t i;
161
162 if(length > 0) {
163 str += length - 1;
164 for(i = 0; i < length; ++i) {
165 if(snmp_packet->used == snmp_packet->max) {
166 return 0;
167 }
168
169 *snmp_packet->out-- = (uint8_t)*str--;
170 snmp_packet->used++;
171 }
172 }
173
174 if(!snmp_ber_encode_length(snmp_packet, length)) {
175 return 0;
176 }
177
178 if(!snmp_ber_encode_type(snmp_packet, BER_DATA_TYPE_OCTET_STRING)) {
179 return 0;
180 }
181
182 return 1;
183 }
184 /*---------------------------------------------------------------------------*/
185 int
186 snmp_ber_encode_oid(snmp_packet_t *snmp_packet, snmp_oid_t *oid)
187 {
188 uint32_t val;
189 uint16_t original_out_len;
190 uint8_t pos;
191
192 original_out_len = snmp_packet->used;
193
194 pos = oid->length - 1;
195 while(pos) {
196 val = oid->data[pos];
197
198 if(snmp_packet->used == snmp_packet->max) {
199 return 0;
200 }
201
202 *snmp_packet->out-- = (uint8_t)(val & 0x7F);
203 snmp_packet->used++;
204 val >>= 7;
205
206 while(val) {
207 if(snmp_packet->used == snmp_packet->max) {
208 return 0;
209 }
210
211 *snmp_packet->out-- = (uint8_t)((val & 0x7F) | 0x80);
212 snmp_packet->used++;
213
214 val >>= 7;
215 }
216 pos--;
217 }
218
219 if(snmp_packet->used == snmp_packet->max) {
220 return 0;
221 }
222
223 // If the given length is zero, and the pointer is at the start,
224 // We will get an out of bounds error here!
225
226 if (snmp_packet->used == 0) {
227 return 0;
228 }
229
230 val = *(snmp_packet->out + 1) + 40 * oid->data[pos]; // This line is bad!
231 snmp_packet->used--;
232 snmp_packet->out++;
233
234 if(snmp_packet->used == snmp_packet->max) {
235 return 0;
236 }
237
238 *snmp_packet->out-- = (uint8_t)(val & 0x7F);
239 snmp_packet->used++;
240
241 val >>= 7;
242
243 while(val) {
244 if(snmp_packet->used == snmp_packet->max) {
245 return 0;
246 }
247
248 *snmp_packet->out-- = (uint8_t)((val & 0x7F) | 0x80);
249 snmp_packet->used++;
250
251 val >>= 7;
252 }
253
254 if(!snmp_ber_encode_length(snmp_packet, snmp_packet->used - original_out_len)) {
255 return 0;
256 }
257
258 if(!snmp_ber_encode_type(snmp_packet, BER_DATA_TYPE_OBJECT_IDENTIFIER)) {
259 return 0;
260 }
261
262 return 1;
263 }

265 int
266 snmp_ber_encode_null(snmp_packet_t *snmp_packet, uint8_t type)
267 {
268
269 if(snmp_packet->used == snmp_packet->max) {
270 return 0;
271 }
272
273 *snmp_packet->out-- = 0x00;
274 snmp_packet->used++;
275
276 return snmp_ber_encode_type(snmp_packet, type);
277 }
278 /*---------------------------------------------------------------------------*/
279 inline int
280 snmp_ber_decode_unsigned_integer(snmp_packet_t *snmp_packet, uint8_t expected_type, uint32_t *num)
281 {
282 uint8_t i, len, type;
283
284 if(!snmp_ber_decode_type(snmp_packet, &type)) {
285 return 0;
286 }
287
288 if(type != expected_type) {
289 /*
290 * Sanity check
291 * Invalid type in buffer
292 */
293 return 0;
294 }
295
296 if(!snmp_ber_decode_length(snmp_packet, &len)) {
297 return 0;
298 }
299
300 if(len > 4) {
301 /*
302 * Sanity check
303 * It will not fit in the uint32_t
304 */
305 return 0;
306 }
307
308 if(snmp_packet->used == 0) {
309 return 0;
310 }
311
312 *num = (uint32_t)(*snmp_packet->in++ & 0xFF);
313 snmp_packet->used--;
314
315 for(i = 1; i < len; ++i) {
316 *num <<= 8;
317 if(snmp_packet->used == 0) {
318 return 0;
319 }
320 *num |= (uint8_t)(*snmp_packet->in++ & 0xFF);
321 snmp_packet->used--;
322 }
323
324 return 1;
325 }

328 snmp_ber_decode_type(snmp_packet_t *snmp_packet, uint8_t *type)
329 {
330 if(snmp_packet->used == 0) {
331 return 0;
332 }
333
334 *type = *snmp_packet->in++;
335 snmp_packet->used--;
336
337 return 1;
338 }

365 int
366 snmp_ber_decode_string_len_buffer(snmp_packet_t *snmp_packet, const char **str, uint32_t *length)
367 {
368 uint8_t type, i, length_bytes;
369
370 if(!snmp_ber_decode_type(snmp_packet, &type)) {
371 return 0;
372 }
373
374 if(type != BER_DATA_TYPE_OCTET_STRING) {
375 /*
376 * Sanity check
377 * Invalid type in buffer
378 */
379 return 0;
380 }
381
382 if (snmp_packet->used == 0) {
383 return 0;
384 }
385
386 if((*snmp_packet->in & 0x80) == 0) {
387
388 *length = (uint32_t)*snmp_packet->in++;
389 snmp_packet->used--;
390 } else {
391
392 length_bytes = (uint8_t)(*snmp_packet->in++ & 0x7F);
393 snmp_packet->used--;
394
395 if(length_bytes > 4) {
396 /*
397 * Sanity check
398 * It will not fit in the uint32_t
399 */
400 return 0;
401 }
402
403 if(snmp_packet->used == 0) {
404 return 0;
405 }
406
407 *length = (uint32_t)*snmp_packet->in++;
408 snmp_packet->used--;
409
410 for(i = 1; i < length_bytes; ++i) {
411 *length <<= 8;
412
413 if(snmp_packet->used == 0) {
414 return 0;
415 }
416
417 *length |= *snmp_packet->in++;
418 snmp_packet->used--;
419 }
420 }
421
422 *str = (const char *)snmp_packet->in;
423
424 if(snmp_packet->used == 0 || snmp_packet->used < *length) {
425 return 0;
426 }
427
428 snmp_packet->used -= *length;
429 snmp_packet->in += *length;
430
431 return 1;
432 }
433 /*---------------------------------------------------------------------------*/
434 int
435 snmp_ber_decode_oid(snmp_packet_t *snmp_packet, snmp_oid_t *oid)
436 {
437 uint8_t *buf_end, type;
438 uint8_t len, j;
439 div_t first;
440
441 if(!snmp_ber_decode_type(snmp_packet, &type)) {
442 return 0;
443 }
444
445 if(type != BER_DATA_TYPE_OBJECT_IDENTIFIER) {
446 return 0;
447 }
448
449 if(!snmp_ber_decode_length(snmp_packet, &len)) { // Pulls in potentially malicious size
450 return 0;
451 }
452
453 buf_end = snmp_packet->in + len;
454
455 if(snmp_packet->used == 0) {
456 return 0;
457 }
458
459 snmp_packet->used--;
460 first = div(*snmp_packet->in++, 40);
461
462 oid->length = 0; // This init could also be removed
463
464 oid->data[oid->length++] = (uint32_t)first.quot;
465 oid->data[oid->length++] = (uint32_t)first.rem;
466
467 while(snmp_packet->in != buf_end) {
468 if(oid->length >= SNMP_MSG_OID_MAX_LEN) { // This line can be removed
469 return 0;
470 }
471
472 if(snmp_packet->used == 0) {
473 return 0;
474 }
475 oid->data[oid->length] = (uint32_t)(*snmp_packet->in & 0x7F);
476 for(j = 0; j < 4; j++) {
477 snmp_packet->used--;
478 if((*snmp_packet->in++ & 0x80) == 0) {
479 break;
480 }
481
482 if(snmp_packet->used == 0) {
483 return 0;
484 }
485
486 oid->data[oid->length] <<= 7;
487 oid->data[oid->length] |= (*snmp_packet->in & 0x7F);
488 }
489
490 oid->length++;
491 }
492
493 return 1;
494 }
495 /*---------------------------------------------------------------------------*/
496 int
497 snmp_ber_decode_null(snmp_packet_t *snmp_packet)
498 {
499 if(snmp_packet->used == 0) {
500 return 0;
501 }
502
503 snmp_packet->in++;
504 snmp_packet->used--;
505
506 if(snmp_packet->used == 0) {
507 return 0;
508 }
509
510 snmp_packet->in++;
511 snmp_packet->used--;
512
513 return 1;
514 }
